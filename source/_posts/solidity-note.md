| uint	      | 0 - 2**256-1|
| __div	      | /           |
| __unm		  | !           |
| __mod		  | %           |
| __pow	      | ^           |
| __concat	  | ..          |
| __eq        | ==          |
| __lt        | <           |
| __le        | <=          |
| __tostring  | 输出字符串    |
| __call	  | 函数调用      |
| __index	  | 调用索引值    |
| __newindex  | 赋值         |
